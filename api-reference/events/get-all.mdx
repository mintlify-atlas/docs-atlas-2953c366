---

## GET /api/events

Returns a paginated list of all events. This endpoint is publicly accessible and does not require authentication.

<Note>
  Results are paginated with a page size of 8 items. Use the `page` parameter to navigate between pages.
</Note>

## Query parameters

<ParamField query="searchTerm" type="string">
  Filter events by title or description. Returns all events when omitted.
</ParamField>

<ParamField query="page" type="number" default="1">
  Page number to retrieve. Starts at 1.
</ParamField>

## Response fields

<ResponseField name="status" type="boolean" required>
  Indicates whether the request succeeded.
</ResponseField>

<ResponseField name="message" type="string">
  Human-readable result message.
</ResponseField>

<ResponseField name="data" type="object" required>
  <Expandable title="properties">
    <ResponseField name="hasNextPage" type="boolean">
      Whether a next page of results exists.
    </ResponseField>
    <ResponseField name="hasPreviousPage" type="boolean">
      Whether a previous page of results exists.
    </ResponseField>
    <ResponseField name="currentPage" type="number">
      The current page number.
    </ResponseField>
    <ResponseField name="pageSize" type="number">
      Number of items per page (8).
    </ResponseField>
    <ResponseField name="totalItems" type="number">
      Total number of events matching the query.
    </ResponseField>
    <ResponseField name="totalPages" type="number">
      Total number of pages.
    </ResponseField>
    <ResponseField name="items" type="object[]">
      <Expandable title="event properties">
        <ResponseField name="id" type="string">
          Unique identifier (GUID) for the event.
        </ResponseField>
        <ResponseField name="categoryId" type="string">
          GUID of the associated category.
        </ResponseField>
        <ResponseField name="categoryName" type="string">
          Display name of the associated category.
        </ResponseField>
        <ResponseField name="organizerId" type="string">
          ID of the user who organized the event.
        </ResponseField>
        <ResponseField name="organizerName" type="string">
          Full name of the organizer.
        </ResponseField>
        <ResponseField name="organizerRating" type="number">
          Average rating of the organizer.
        </ResponseField>
        <ResponseField name="organizerRatingsCount" type="number">
          Total number of ratings the organizer has received.
        </ResponseField>
        <ResponseField name="title" type="string">
          Title of the event.
        </ResponseField>
        <ResponseField name="description" type="string">
          Full description of the event.
        </ResponseField>
        <ResponseField name="ubication" type="string">
          Physical location or address of the event.
        </ResponseField>
        <ResponseField name="date" type="string">
          Date and time of the event (ISO 8601).
        </ResponseField>
        <ResponseField name="publicationDate" type="string">
          Date and time the event was published (ISO 8601).
        </ResponseField>
        <ResponseField name="attendancesCount" type="number">
          Total number of attendees.
        </ResponseField>
        <ResponseField name="commentsCount" type="number">
          Total number of comments.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
